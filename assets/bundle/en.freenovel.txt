(function () {
  'use strict';

  const BASE = 'https://freewebnovel.com';

  const sleep = ms => new Promise(r => setTimeout(r, ms));

  // ─── HTTP ────────────────────────────────────────────────

  async function fetchText(url, options = {}, retries = 2) {
    try {
      const res = await fetch(url, {
        method: options.method || 'GET',
        headers: {
          'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36',
          'Referer': BASE + '/',
          'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
          'Accept-Language': 'en-US,en;q=0.5',
          ...(options.headers || {}),
        },
        body: options.body,
      });

      if ((res.status === 403 || res.status === 503) && retries > 0) {
        await sleep(1500);
        return fetchText(url, options, retries - 1);
      }

      if (!res.ok) throw new Error(`HTTP ${res.status} — ${url}`);
      return res.text();
    } catch (err) {
      if (retries > 0) {
        await sleep(1000);
        return fetchText(url, options, retries - 1);
      }
      throw err;
    }
  }

  function parseHTML(html) {
    return parse(html);
  }

  function absoluteUrl(href) {
    if (!href) return '';
    return href.startsWith('http') ? href : BASE + href;
  }

  function novelIdFromUrl(url) {
    const m = url.match(/\/novel\/([^/?#]+)/);
    return m ? m[1] : url;
  }

  // ─── LISTING PAGE PARSER ─────────────────────────────────
  //
  // Verified from real HTML:
  //   card      : div.li-row
  //   title+url : h3.tit a
  //   cover     : .pic img  [src]
  //   rating    : .core span
  //   language  : .desc .item em.e1 a
  //   genres    : .desc .item a.novel[href*="/genre/"]
  //   chapters  : .desc .item a.chapter span.s1  (e.g. "30 Chapters")
  //   latest ch : .desc .item a.chapter  [href]
  //   completed : span.s2  (text "Full")

  function parseListingCards(root) {
    const results = [];
    const seen = new Set();

    const cards = root.querySelectorAll('div.li-row');
    cards.forEach(card => {
      const titleEl = card.querySelector('h3.tit a');
      const href = titleEl?.getAttribute('href');
      if (!href) return;

      const url = absoluteUrl(href);
      if (!url || seen.has(url)) return;
      seen.add(url);

      const imgEl = card.querySelector('.pic img');
      const coverUrl = absoluteUrl(
        imgEl?.getAttribute('src') ?? imgEl?.getAttribute('data-src') ?? ''
      );

      const genres = card
        .querySelectorAll('.desc .item a.novel[href*="/genre/"]')
        .map(a => a.text.trim())
        .filter(Boolean);

      const chapEl = card.querySelector('.desc .item a.chapter');
      const chapCountText = chapEl?.querySelector('span.s1')?.text.trim() ?? '';
      const chapUrl = chapEl ? absoluteUrl(chapEl.getAttribute('href')) : '';

      const fullBadge = card.querySelector('span.s2');
      const completed = fullBadge?.text.trim() === 'Full';

      results.push({
        id: novelIdFromUrl(href),
        sourceId: 'en.freewebnovel',
        title: titleEl.text.trim(),
        coverUrl,
        url,
        genres,
        status: completed ? 'completed' : 'ongoing',
        chapCountText,
        latestChapterUrl: chapUrl,
      });
    });

    return results;
  }

  // ─── SEARCH / POPULAR ────────────────────────────────────
  //
  // Search  : POST /search/          body: searchkey=<query>
  //           page 2+: POST /search/2/
  //
  // Popular : GET  /sort/most-popular/     (page 1)
  //           GET  /sort/most-popular/2/   (page 2+)
  //
  // Pagination: div.pages ul li a[href]
  //   href pattern: /sort/most-popular/2, /sort/most-popular/3 …
  //   current page: div.pages ul li strong  (no href)

  async function search(query, page) {
    let html;
    if (query && query.trim()) {
      const pageSegment = page > 1 ? `${page}/` : '';
      html = await fetchText(`${BASE}/search/${pageSegment}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: `searchkey=${encodeURIComponent(query.trim())}`,
      });
    } else {
      const pageSegment = page > 1 ? `${page}/` : '';
      html = await fetchText(`${BASE}/sort/most-popular/${pageSegment}`);
    }

    return parseListingCards(parseHTML(html));
  }

  async function getPopular(page) {
    return search('', page);
  }

  // ─── NOVEL DETAIL ────────────────────────────────────────
  //
  // Verified selectors:
  //   title       : h1.tit
  //   cover       : .pic img
  //   author      : .txt .item a.a1
  //   genres      : .txt .item a[href*="/genre/"]
  //   language    : em.e1 a  (or a[href*="english-novel"])
  //   status      : .item .s1.s2 a   → "OnGoing" / "Completed"
  //   description : .m-desc .inner
  //   latest chs  : .m-newest1 .ul-list5 li a.con  (top 6)

  async function getNovelDetail(url) {
    const html = await fetchText(url);
    const root = parseHTML(html);

    const title = root.querySelector('h1.tit')?.text.trim() ?? '';

    const imgEl = root.querySelector('.pic img');
    const coverUrl = absoluteUrl(
      imgEl?.getAttribute('src') ?? imgEl?.getAttribute('data-src') ?? ''
    );

    const author = root.querySelector('.txt .item a.a1')?.text.trim() ?? '';

    const genres = root
      .querySelectorAll('.txt .item a[href*="/genre/"]')
      .map(a => a.text.trim())
      .filter(Boolean);

    // Language
    let language = 'en';
    const langHref = root.querySelector('em.e1 a')?.getAttribute('href') ?? '';
    if (langHref.includes('chinese')) language = 'zh';
    else if (langHref.includes('korean')) language = 'ko';

    // Status
    const statusRaw = root.querySelector('.item .s1.s2 a')?.text.trim().toLowerCase() ?? '';
    const status = statusRaw.includes('ongoing')  ? 'ongoing'
      : statusRaw.includes('complet') ? 'completed'
      : statusRaw.includes('hiatus')  ? 'hiatus'
      : 'unknown';

    const description = root.querySelector('.m-desc .inner')?.text.trim() ?? '';

    // Latest chapter (first of top-6 list)
    const latestEl = root.querySelector('.m-newest1 .ul-list5 li a.con');
    const latestChapter = latestEl
      ? { title: latestEl.text.trim(), url: absoluteUrl(latestEl.getAttribute('href')) }
      : null;

    return {
      id: novelIdFromUrl(url),
      sourceId: 'en.freewebnovel',
      title,
      coverUrl,
      url,
      authors: author ? [author] : [],
      genres,
      language,
      status,
      description,
      latestChapter,
    };
  }

  // ─── CHAPTER LIST ────────────────────────────────────────
  //
  // All chapters are in the HTML already (hidden per group via display:none).
  // Verified selector: .m-newest2 #idData li a.con
  // No JS execution needed — all <li> tags present in source.

  async function getChapters(novelUrl) {
    const html = await fetchText(novelUrl);
    const root = parseHTML(html);

    let links = root.querySelectorAll('.m-newest2 #idData li a.con');
    if (!links.length) links = root.querySelectorAll('.m-newest2 ul li a');
    if (!links.length) links = root.querySelectorAll('#idData li a');
    if (!links.length) links = root.querySelectorAll('a[href*="/chapter-"]');

    const chapters = [];
    const seen = new Set();

    links.forEach((el, index) => {
      const href = el.getAttribute('href');
      if (!href || seen.has(href)) return;
      seen.add(href);

      const url = absoluteUrl(href);
      const numMatch = href.match(/chapter-([\d]+(?:\.\d+)?)/i);
      const number = numMatch ? parseFloat(numMatch[1]) : index + 1;
      const id = href
        .replace(/^\//, '')
        .replace(/\//g, '-')
        .replace(/\.html$/, '');

      chapters.push({
        id,
        title: el.text.trim() || `Chapter ${number}`,
        number,
        url,
      });
    });

    chapters.sort((a, b) => a.number - b.number);
    return chapters;
  }

  // ─── CHAPTER CONTENT ────────────────────────────────────
  //
  // Verified selectors:
  //   content container : div.txt #article
  //   paragraphs        : p  (inside #article)
  //   watermark tags    : subtxt  → strip before extracting text
  //   prev chapter      : a#prev_url  — points to novel page when on chapter 1
  //   next chapter      : a#next_url  — points to novel page when on last chapter
  //
  // Detection: if href contains "/novel/" but NOT "/chapter-" → boundary (first/last)

  function isChapterUrl(href) {
    return href && href.includes('/chapter-');
  }

  async function getChapterContent(chapterUrl) {
    const html = await fetchText(chapterUrl);
    const root = parseHTML(html);

    const article =
      root.querySelector('div.txt #article') ??
      root.querySelector('#article') ??
      root.querySelector('.chapter-content') ??
      root.querySelector('#chapter-content');

    if (!article) return [{ index: 0, text: '' }];

    // Strip watermark <subtxt> nodes — replace their raw HTML with empty string
    // node-html-parser: use innerHTML manipulation to remove subtxt
    const paras = article.querySelectorAll('p');
    const result = [];
    let idx = 0;

    paras.forEach(p => {
      // Remove all subtxt children by rebuilding text without them
      let text = '';
      p.childNodes.forEach(node => {
        // TextNode has nodeType 3; element nodes have tagName
        if (!node.tagName) {
          // raw text node
          text += node.text ?? '';
        } else if (node.tagName.toLowerCase() !== 'subtxt') {
          text += node.text ?? '';
        }
        // subtxt nodes are skipped entirely
      });

      text = text.trim();
      if (text.length > 0) result.push({ index: idx++, text });
    });

    if (result.length > 0) return result;

    // Fallback: strip subtxt via regex on raw text
    return article.text
      .split('\n')
      .map(t => t.trim())
      .filter(t => t.length > 0)
      .map((text, index) => ({ index, text }));
  }

  // ─── REGISTER ───────────────────────────────────────────

  __xaina_register({
    id: 'en.freewebnovel',
    name: 'FreeWebNovel',
    lang: 'en',
    version: '9.3.0',
    baseUrl: BASE,
    iconUrl: `${BASE}/favicon.ico`,
    search,
    getPopular,
    getNovelDetail,
    getChapters,
    getChapterContent,
  });

})();
