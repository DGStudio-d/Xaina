(function () {
  'use strict';

  var BASE = 'https://freewebnovel.com';

  function sleep(ms) { return new Promise(function(r) { setTimeout(r, ms); }); }

  // ─── HTTP ────────────────────────────────────────────────

  async function fetchText(url, options, retries) {
    if (!options) options = {};
    if (retries === undefined) retries = 2;
    try {
      var res = await fetch(url, {
        method: options.method || 'GET',
        headers: Object.assign({
          'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36',
          'Referer': BASE + '/',
          'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
          'Accept-Language': 'en-US,en;q=0.5',
        }, options.headers || {}),
        body: options.body,
      });
      if ((res.status === 403 || res.status === 503) && retries > 0) {
        await sleep(1500);
        return fetchText(url, options, retries - 1);
      }
      if (!res.ok) throw new Error('HTTP ' + res.status + ' - ' + url);
      return res.text();
    } catch (err) {
      if (retries > 0) { await sleep(1000); return fetchText(url, options, retries - 1); }
      throw err;
    }
  }

  // Parse options — skip unnecessary processing for speed
  var PARSE_OPTS = {
    lowerCaseTagName: false,
    comment: false,
    fixNestedATags: false,
    parseNoneClosedTags: false,
    blockTextElements: {
      script: false,  // don't parse script content
      noscript: false,
      style: false,   // don't parse style content
      pre: true,
    },
  };

  function parseHTML(html) { return parse(html, PARSE_OPTS); }

  // Extract only the content section before parsing — avoids parsing nav/footer/scripts
  function parseSection(html, selector) {
    var tagMatch = selector.match(/^([a-z]+)/i);
    var classMatch = selector.match(/\.([a-z0-9_-]+)/i);
    if (tagMatch && classMatch) {
      var tag = tagMatch[1];
      var cls = classMatch[1];
      var re = new RegExp('<' + tag + '[^>]*class="[^"]*' + cls + '[^"]*"[\\s\\S]*', 'i');
      var m = html.match(re);
      if (m) return parse(m[0], PARSE_OPTS);
    }
    return parse(html, PARSE_OPTS);
  }

  function abs(href) {
    if (!href) return '';
    return href.startsWith('http') ? href : BASE + href;
  }

  function novelId(url) {
    var m = url.match(/\/novel\/([^/?#]+)/);
    return m ? m[1] : url;
  }

  // ─── LISTING PARSER ──────────────────────────────────────
  // Verified HTML structure:
  //   div.li-row > div.li > div.con
  //     div.pic img          → cover
  //     h3.tit a             → title + href
  //     em.e1 a              → language
  //     .desc .item a.novel[href*="/genre/"]  → genres
  //     a.chapter span.s1    → chapter count text
  //     a.chapter span.s2    → "Full" badge
  //     div.pages ul li a    → pagination links

  function parseCards(root) {
    var results = [];
    var seen = new Set();

    var rows = root.querySelectorAll('div.li-row');
    rows.forEach(function(row) {
      var titleEl = row.querySelector('h3.tit a');
      if (!titleEl) return;
      var href = titleEl.getAttribute('href');
      if (!href) return;
      var url = abs(href);
      if (!url || seen.has(url)) return;
      seen.add(url);

      var title = titleEl.text.trim();
      if (!title) return;

      var imgEl = row.querySelector('div.pic img');
      var coverUrl = abs(imgEl ? (imgEl.getAttribute('src') || '') : '');

      var genres = row.querySelectorAll('.desc .item a.novel[href*="/genre/"]')
        .map(function(a) { return a.text.trim(); })
        .filter(Boolean);

      var s2 = row.querySelector('a.chapter span.s2');
      var status = (s2 && s2.text.trim() === 'Full') ? 'completed' : 'ongoing';

      results.push({
        id: novelId(href),
        sourceId: 'en.freewebnovel',
        title: title,
        coverUrl: coverUrl,
        url: url,
        genres: genres,
        status: status,
      });
    });

    return results;
  }

  // ─── SEARCH / POPULAR ────────────────────────────────────
  // Popular : /sort/most-popular  (no pagination)
  // Search  : POST /search/ (p1), /search/2 (p2+)
  //           paginated via div.pages ul li a[href]

  async function search(query, page) {
    var html;
    if (query && query.trim()) {
      var seg = page > 1 ? ('' + page) : '';
      html = await fetchText(BASE + '/search/' + seg, {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: 'searchkey=' + encodeURIComponent(query.trim()),
      });
    } else {
      if (page > 1) return [];
      html = await fetchText(BASE + '/sort/most-popular');
    }
    // Parse only the content column — skips nav, sidebar, footer, scripts
    return parseCards(parseSection(html, 'div.col-content'));
  }

  async function getPopular(page) { return search('', page); }

  // ─── NOVEL DETAIL ────────────────────────────────────────
  // Verified selectors from live HTML:
  //   h3.tit or h1          → title
  //   div.pic img            → cover
  //   a[href*="/author/"]    → author
  //   .desc .item a.novel[href*="/genre/"]  → genres
  //   em.e1 a                → language
  //   a[href*="latest-release"] text → status (OnGoing/Completed)
  //   .m-desc                → description
  //   a[href*="/chapter-"]   → latest chapter (first match)

  async function getNovelDetail(url) {
    var html = await fetchText(url);
    var root = parseHTML(html);

    var h1 = root.querySelector('h1');
    var title = h1 ? h1.text.trim() : '';

    var imgEl = root.querySelector('div.pic img') || root.querySelector('img[src*="/files/article/image/"]');
    var coverUrl = abs(imgEl ? (imgEl.getAttribute('src') || '') : '');

    var authorEl = root.querySelector('a[href*="/author/"]');
    var author = authorEl ? authorEl.text.trim() : '';

    var genres = root.querySelectorAll('.desc .item a.novel[href*="/genre/"]')
      .map(function(a) { return a.text.trim(); }).filter(Boolean);

    var language = 'en';
    var e1 = root.querySelector('em.e1 a');
    if (e1) {
      var lh = e1.getAttribute('href') || '';
      if (lh.includes('chinese')) language = 'zh';
      else if (lh.includes('korean')) language = 'ko';
      else if (lh.includes('japanese')) language = 'ja';
    }

    // Status from the sort link text (OnGoing / Completed)
    var statusEl = root.querySelector('a[href*="/sort/latest-release"]');
    var statusRaw = statusEl ? statusEl.text.trim().toLowerCase() : '';
    var status = statusRaw.includes('ongoing') ? 'ongoing'
      : statusRaw.includes('complet') ? 'completed'
      : 'unknown';

    var descEl = root.querySelector('.m-desc');
    var description = descEl ? descEl.text.trim() : '';

    var slug = novelId(url);
    var chLinks = root.querySelectorAll('a[href*="/novel/' + slug + '/chapter-"]');
    var latestEl = chLinks[0] || null;
    var latestChapter = latestEl
      ? { title: latestEl.text.trim(), url: abs(latestEl.getAttribute('href')) }
      : null;

    return {
      id: slug,
      sourceId: 'en.freewebnovel',
      title: title,
      coverUrl: coverUrl,
      url: url,
      authors: author ? [author] : [],
      genres: genres,
      language: language,
      status: status,
      description: description,
      latestChapter: latestChapter,
    };
  }

  // ─── CHAPTER LIST ────────────────────────────────────────
  // All chapter links are in the HTML as a[href*="/novel/slug/chapter-"]

  async function getChapters(novelUrl) {
    var html = await fetchText(novelUrl);
    var root = parseHTML(html);
    var slug = novelId(novelUrl);
    var links = root.querySelectorAll('a[href*="/novel/' + slug + '/chapter-"]');

    var chapters = [];
    var seen = new Set();

    links.forEach(function(el, index) {
      var href = el.getAttribute('href');
      if (!href || seen.has(href)) return;
      seen.add(href);
      var url = abs(href);
      var m = href.match(/chapter-([\d]+(?:\.\d+)?)/i);
      var number = m ? parseFloat(m[1]) : index + 1;
      var id = href.replace(/^\//, '').replace(/\//g, '-').replace(/\.html$/, '');
      chapters.push({ id: id, title: el.text.trim() || ('Chapter ' + number), number: number, url: url });
    });

    chapters.sort(function(a, b) { return a.number - b.number; });
    return chapters;
  }

  // ─── CHAPTER CONTENT ────────────────────────────────────
  // Container: div.txt #article or #article
  // Strip <subtxt> watermark nodes

  async function getChapterContent(chapterUrl) {
    var html = await fetchText(chapterUrl);
    var root = parseHTML(html);

    var article = root.querySelector('div.txt #article')
      || root.querySelector('#article')
      || root.querySelector('.chapter-content')
      || root.querySelector('#chapter-content');

    if (!article) return [{ index: 0, text: '' }];

    var paras = article.querySelectorAll('p');
    var result = [];
    var idx = 0;

    paras.forEach(function(p) {
      var text = '';
      p.childNodes.forEach(function(node) {
        if (!node.tagName) { text += node.text || ''; }
        else if (node.tagName.toLowerCase() !== 'subtxt') { text += node.text || ''; }
      });
      text = text.trim();
      if (text.length > 0) result.push({ index: idx++, text: text });
    });

    if (result.length > 0) return result;

    return article.text.split('\n')
      .map(function(t) { return t.trim(); })
      .filter(function(t) { return t.length > 0; })
      .map(function(text, i) { return { index: i, text: text }; });
  }

  // ─── REGISTER ───────────────────────────────────────────

  __xaina_register({
    id: 'en.freewebnovel',
    name: 'FreeWebNovel',
    lang: 'en',
    version: '12.1.0',
    baseUrl: BASE,
    iconUrl: BASE + '/favicon.ico',
    search: search,
    getPopular: getPopular,
    getNovelDetail: getNovelDetail,
    getChapters: getChapters,
    getChapterContent: getChapterContent,
  });

})();
