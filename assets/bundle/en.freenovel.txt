(function () {
  'use strict';

  const BASE = 'https://freewebnovel.com';

  const sleep = ms => new Promise(r => setTimeout(r, ms));

  // ─── HTTP ────────────────────────────────────────────────

  async function fetchText(url, options = {}, retries = 2) {
    try {
      const res = await fetch(url, {
        method: options.method || 'GET',
        headers: {
          'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36',
          'Referer': BASE + '/',
          'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
          'Accept-Language': 'en-US,en;q=0.5',
          ...(options.headers || {}),
        },
        body: options.body,
      });

      if ((res.status === 403 || res.status === 503) && retries > 0) {
        await sleep(1500);
        return fetchText(url, options, retries - 1);
      }

      if (!res.ok) throw new Error(`HTTP ${res.status} — ${url}`);
      return res.text();
    } catch (err) {
      if (retries > 0) {
        await sleep(1000);
        return fetchText(url, options, retries - 1);
      }
      throw err;
    }
  }

  function parseHTML(html) {
    return parse(html);
  }

  function absoluteUrl(href) {
    if (!href) return '';
    return href.startsWith('http') ? href : BASE + href;
  }

  function novelIdFromUrl(url) {
    const m = url.match(/\/novel\/([^/?#]+)/);
    return m ? m[1] : url;
  }

  // ─── LISTING PAGE PARSER ─────────────────────────────────
  //
  // Live HTML structure (verified):
  //   Each entry has: img, h3 > a (title+href), language link, genre links,
  //   chapter count link, "Full" badge
  //
  //   Cover img src: /files/article/image/N/NNN/NNNss.jpg
  //   Title + href : h3 a  (inside the card section)
  //   Language     : a[href*="latest-release"]
  //   Genres       : a[href*="/genre/"]
  //   Chapter link : a[href*="/chapter-"]  (last chapter URL + count text)
  //   Full badge   : text "Full" or "HOT" or "HOTFULL" before the h3

  function parseListingCards(root) {
    const results = [];
    const seen = new Set();

    // Each novel entry has an h3 with an anchor — use that as the anchor point
    const h3s = root.querySelectorAll('h3 a[href*="/novel/"]');

    h3s.forEach(titleEl => {
      const href = titleEl.getAttribute('href');
      if (!href) return;
      const url = absoluteUrl(href);
      if (!url || seen.has(url)) return;
      seen.add(url);

      const title = titleEl.text.trim();
      if (!title) return;

      // Walk up to find the containing section — look for nearby img
      // The cover img has src pattern /files/article/image/...
      // It appears before the h3 in the same block
      // We'll extract the novel slug from href and build the cover URL
      const slug = novelIdFromUrl(href);

      // Try to find cover from nearby img — search in parent nodes
      let coverUrl = '';
      let node = titleEl.parentNode; // h3
      // Walk up a few levels to find an img sibling
      for (let i = 0; i < 5 && node; i++) {
        const img = node.querySelector('img[src*="/files/article/image/"]');
        if (img) {
          coverUrl = absoluteUrl(img.getAttribute('src') ?? '');
          break;
        }
        node = node.parentNode;
      }

      // Genres: links to /genre/ near this h3
      const genres = [];
      let gNode = titleEl.parentNode;
      for (let i = 0; i < 5 && gNode; i++) {
        gNode.querySelectorAll('a[href*="/genre/"]').forEach(a => {
          const g = a.text.trim();
          if (g) genres.push(g);
        });
        if (genres.length) break;
        gNode = gNode.parentNode;
      }

      // Status: look for "Full" text near the h3
      let status = 'ongoing';
      let sNode = titleEl.parentNode;
      for (let i = 0; i < 5 && sNode; i++) {
        if (sNode.text && sNode.text.includes('Full')) { status = 'completed'; break; }
        sNode = sNode.parentNode;
      }

      results.push({
        id: slug,
        sourceId: 'en.freewebnovel',
        title,
        coverUrl,
        url,
        genres,
        status,
      });
    });

    return results;
  }

  // ─── SEARCH / POPULAR ────────────────────────────────────

  async function search(query, page) {
    let html;
    if (query && query.trim()) {
      const pageSegment = page > 1 ? `${page}/` : '';
      html = await fetchText(`${BASE}/search/${pageSegment}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: `searchkey=${encodeURIComponent(query.trim())}`,
      });
    } else {
      const pageSegment = page > 1 ? `${page}/` : '';
      html = await fetchText(`${BASE}/sort/most-popular/${pageSegment}`);
    }

    return parseListingCards(parseHTML(html));
  }

  async function getPopular(page) {
    return search('', page);
  }

  // ─── NOVEL DETAIL ────────────────────────────────────────
  //
  // Live HTML (verified from shadow-slave page):
  //   Cover   : img[src*="/files/article/image/"] (first one, the large cover)
  //   Author  : a[href*="/author/"]
  //   Genres  : a[href*="/genre/"]
  //   Language: a[href*="latest-release/english-novel"] etc.
  //   Status  : text "OnGoing" or "Completed" — look for a[href*="latest-release"]:not([href*="novel"])
  //   Summary : section after "SUMMARY" heading — .m-desc or look for the paragraph block
  //   Latest  : links in "6 LATEST CHAPTERS" section — a[href*="/chapter-"] (first 6)
  //   Chapters: all a[href*="/chapter-"] that match /novel/slug/chapter-N

  async function getNovelDetail(url) {
    const html = await fetchText(url);
    const root = parseHTML(html);

    const title = root.querySelector('h1')?.text.trim()
      ?? root.querySelector('h3')?.text.trim()
      ?? '';

    // Cover: first large image
    const imgEl = root.querySelector('img[src*="/files/article/image/"]');
    const coverUrl = absoluteUrl(
      imgEl?.getAttribute('src') ?? imgEl?.getAttribute('data-src') ?? ''
    );

    // Author
    const author = root.querySelector('a[href*="/author/"]')?.text.trim() ?? '';

    // Genres
    const genres = root
      .querySelectorAll('a[href*="/genre/"]')
      .map(a => a.text.trim())
      .filter(Boolean);

    // Language
    let language = 'en';
    const langEl = root.querySelector('a[href*="latest-release/chinese"]')
      ?? root.querySelector('a[href*="latest-release/korean"]')
      ?? root.querySelector('a[href*="latest-release/japanese"]');
    if (langEl) {
      const h = langEl.getAttribute('href') ?? '';
      if (h.includes('chinese')) language = 'zh';
      else if (h.includes('korean')) language = 'ko';
      else if (h.includes('japanese')) language = 'ja';
    }

    // Status: look for OnGoing / Completed text
    const bodyText = root.text;
    const status = /completed/i.test(bodyText) && !/ongoing/i.test(bodyText)
      ? 'completed'
      : /ongoing/i.test(bodyText) ? 'ongoing' : 'unknown';

    // Description: the SUMMARY section paragraphs
    const description = root.querySelector('.m-desc')?.text.trim()
      ?? root.querySelector('.summary')?.text.trim()
      ?? '';

    // Latest chapter: first chapter link in the "LATEST CHAPTERS" block
    const slug = novelIdFromUrl(url);
    const chapterLinks = root.querySelectorAll(`a[href*="/novel/${slug}/chapter-"]`);
    const latestEl = chapterLinks[0] ?? null;
    const latestChapter = latestEl
      ? { title: latestEl.text.trim(), url: absoluteUrl(latestEl.getAttribute('href')) }
      : null;

    return {
      id: slug,
      sourceId: 'en.freewebnovel',
      title,
      coverUrl,
      url,
      authors: author ? [author] : [],
      genres,
      language,
      status,
      description,
      latestChapter,
    };
  }

  // ─── CHAPTER LIST ────────────────────────────────────────
  //
  // Live HTML (verified): chapter links are plain anchors
  //   a[href*="/novel/slug/chapter-"]
  // All chapters visible in HTML (JS pagination hides groups via display:none
  // but all <a> tags are present in the source).

  async function getChapters(novelUrl) {
    const html = await fetchText(novelUrl);
    const root = parseHTML(html);

    const slug = novelIdFromUrl(novelUrl);
    const links = root.querySelectorAll(`a[href*="/novel/${slug}/chapter-"]`);

    const chapters = [];
    const seen = new Set();

    links.forEach((el, index) => {
      const href = el.getAttribute('href');
      if (!href || seen.has(href)) return;
      seen.add(href);

      const url = absoluteUrl(href);
      const numMatch = href.match(/chapter-([\d]+(?:\.\d+)?)/i);
      const number = numMatch ? parseFloat(numMatch[1]) : index + 1;
      const id = href
        .replace(/^\//, '')
        .replace(/\//g, '-')
        .replace(/\.html$/, '');

      chapters.push({
        id,
        title: el.text.trim() || `Chapter ${number}`,
        number,
        url,
      });
    });

    chapters.sort((a, b) => a.number - b.number);
    return chapters;
  }

  // ─── CHAPTER CONTENT ────────────────────────────────────
  //
  // Live HTML (verified):
  //   Content container : div.txt #article  or  #article
  //   Paragraphs        : p tags inside
  //   Watermarks        : <subtxt> tags inside <p> — skip their text

  async function getChapterContent(chapterUrl) {
    const html = await fetchText(chapterUrl);
    const root = parseHTML(html);

    const article =
      root.querySelector('div.txt #article') ??
      root.querySelector('#article') ??
      root.querySelector('.chapter-content') ??
      root.querySelector('#chapter-content');

    if (!article) return [{ index: 0, text: '' }];

    const paras = article.querySelectorAll('p');
    const result = [];
    let idx = 0;

    paras.forEach(p => {
      let text = '';
      p.childNodes.forEach(node => {
        if (!node.tagName) {
          text += node.text ?? '';
        } else if (node.tagName.toLowerCase() !== 'subtxt') {
          text += node.text ?? '';
        }
      });
      text = text.trim();
      if (text.length > 0) result.push({ index: idx++, text });
    });

    if (result.length > 0) return result;

    return article.text
      .split('\n')
      .map(t => t.trim())
      .filter(t => t.length > 0)
      .map((text, index) => ({ index, text }));
  }

  // ─── REGISTER ───────────────────────────────────────────

  __xaina_register({
    id: 'en.freewebnovel',
    name: 'FreeWebNovel',
    lang: 'en',
    version: '10.1.0',
    baseUrl: BASE,
    iconUrl: `${BASE}/favicon.ico`,
    search,
    getPopular,
    getNovelDetail,
    getChapters,
    getChapterContent,
  });

})();
